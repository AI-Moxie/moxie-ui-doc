static ButtonStyle styleFrom({
  TextStyle? textStyle,
  Color? foregroundColor = const Color(0xFF0469B5),
  Color? disabledForegroundColor = const Color(0xFF888888),
  Color? backgroundColor = const Color(0xFFFFFFFF),
  Color? disabledBackgroundColor = const Color(0xFFCCCCCC),
  Color? overlayColor = Colors.transparent,
  Color? shadowColor,
  double? elevation,
  AlignmentGeometry? alignment,
  EdgeInsetsGeometry? padding = const EdgeInsets.symmetric(
    horizontal: 8.0,
    vertical: 4.0,
  ),
  Size? size,
  BorderSide? side = const BorderSide(
    color: Color(0xFF0469B5),
    width: 1,
  ),
  OutlinedBorder? shape = const RoundedRectangleBorder(
    borderRadius: BorderRadius.all(Radius.circular(10)),
  ),
  VisualDensity? visualDensity,
  MaterialTapTargetSize? tapTargetSize,
  InteractiveInkFeatureFactory? splashFactory = NoSplash.splashFactory,
}) {
  MaterialStateProperty<Color?> foregroundColorProperty =
      _OutlinedButtonColor(foregroundColor, disabledForegroundColor);
  MaterialStateProperty<Color?> backgroundColorProperty =
      _OutlinedButtonColor(backgroundColor, disabledBackgroundColor);

  return ButtonStyle(
    textStyle: ButtonStyleButton.allOrNull<TextStyle?>(textStyle),
    foregroundColor: foregroundColorProperty,
    backgroundColor: backgroundColorProperty,
    overlayColor: ButtonStyleButton.allOrNull<Color?>(overlayColor),
    shadowColor: ButtonStyleButton.allOrNull<Color?>(shadowColor),
    elevation: ButtonStyleButton.allOrNull<double?>(elevation),
    alignment: alignment,
    padding: ButtonStyleButton.allOrNull<EdgeInsetsGeometry?>(padding),
    minimumSize: ButtonStyleButton.allOrNull<Size?>(size),
    side: ButtonStyleButton.allOrNull<BorderSide?>(side),
    shape: ButtonStyleButton.allOrNull<OutlinedBorder?>(shape),
    visualDensity: visualDensity,
    tapTargetSize: tapTargetSize,
    splashFactory: splashFactory,
  );
}
